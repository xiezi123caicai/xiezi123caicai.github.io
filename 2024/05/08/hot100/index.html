class Solution {

            String code = encode(s);
            codeToGroup.putIfAbsent(code, new LinkedList<>());
            codeToGroup.get(code).add(s);
        }

        }

    private String encode(String s){

            count[c - 'a']++;
        }
        return new String(count);
    }
